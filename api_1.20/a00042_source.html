00067 {
00068     int d1;
00069     int d2;
00070     int newcount;
00071 
00072     if(set1 == set2)
00073         return set1;
00074 
00075     /* need 2 set representatives */
00076     d1 = data[set1];
00077     d2 = data[set2];

00079 
00080     newcount = d1 + d2;
