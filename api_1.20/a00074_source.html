00245 #define hook_exec(what, args) do {           \
00246   hook_entry_t *_p;                          \
00247   for (_p = hooks[what]; _p; _p = _p->next){ \
00248     void *hook_ctx_ = _p->context;           \
00249     _p->hook._##what args;                   \
00250   }                                          \
00251 } while (0)
