241 #define hook_exec(what, args) do { \
242  hook_entry_t *_p; \
243  for (_p = hooks[what]; _p; _p = _p->next){ \
244  void *hook_ctx_ = _p->context; \
245  _p->hook._##what args; \
246  } \
247 } while (0)
